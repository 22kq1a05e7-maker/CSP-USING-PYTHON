problem-2 Calculator
a = float(input("Enter first: ")) 
b = float(input("Enter second: ")) 
op = input("Enter operation (+,-,,/): ")
if op == '+': print(a + b) 
elif op == '-': print(a - b) 
elif op == '*': print(a * b) 
elif op == '/': print(a / b) 
else: print("Invalid operator")
